链式编程思想

链式编程思想：把一系列的操作使用.来来连接成一句代码，可读性好。典型的代表Masonry框架，例如：

[redView mas_makeConstraints:^(MASConstraintMaker *make) {
    make.left.top.equalTo(@10);
    make.right.bottom.equalTo(@(-10));
}];

ZHCalculateMaker的声明与实现

#import <Foundation/Foundation.h>

@interface ZHCalculateMaker : NSObject

@property (nonatomic, assign) float result;

- (ZHCalculateMaker * (^)(float))add;

- (ZHCalculateMaker * (^)(float))subtract;

- (ZHCalculateMaker * (^)(float))multiply;

- (ZHCalculateMaker * (^)(float))divide;

@end

#import "ZHCalculateMaker.h"

@implementation ZHCalculateMaker

- (ZHCalculateMaker * (^)(float))add {
    return ^(float num) {
        self.result += num;
        return self;
    };
}

- (ZHCalculateMaker * (^)(float))subtract {
    return ^(float num) {
        self.result -= num;
        return self;
    };
}

- (ZHCalculateMaker * (^)(float))multiply {
    return ^(float num) {
        self.result *= num;
        return self;
    };
}

- (ZHCalculateMaker * (^)(float))divide {
    return ^(float num) {
        self.result /= num;
        return self;
    };
}

@end

NSObject+ZHCalculate.h的声明与实现

#import <Foundation/Foundation.h>
#import "ZHCalculateMaker.h"

@interface NSObject (ZHCalculate)

+ (float)makeCalculate:(void (^)(ZHCalculateMaker *maker))block;

@end

#import "NSObject+ZHCalculate.h"
#import "ZHCalculateMaker.h"

@implementation NSObject (ZHCalculate)

+ (float)makeCalculate:(void (^)(ZHCalculateMaker *maker))block {
    //初始化制造者
    ZHCalculateMaker *maker = [ZHCalculateMaker new];

    block(maker);

    return maker.result;
}

@end

外部调用方法：

- (void)viewDidLoad {
    [super viewDidLoad];

    float result = [NSObject makeCalculate:^(ZHCalculateMaker *maker) {

        maker.add(3).multiply(5);

    }];

    NSLog(@"%f", result);
}
